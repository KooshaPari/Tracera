# Webhooks

Webhooks deliver real-time HTTP notifications when events occur in TraceRTM. Use webhooks to integrate with external systems, trigger automation workflows, and build custom notifications.

## Quick Reference

```bash
# List webhooks
GET /v1/webhooks

# Create webhook
POST /v1/webhooks

# Manage webhook
GET/PATCH/DELETE /v1/webhooks/{webhook_id}

# Test and deliveries
POST /v1/webhooks/{webhook_id}/test
GET /v1/webhooks/{webhook_id}/deliveries
```

---

## How Webhooks Work

```
1. Event occurs in TraceRTM (e.g., item approved)
          ↓
2. TraceRTM checks for matching webhook subscriptions
          ↓
3. Payload is constructed with event data
          ↓
4. HTTP POST sent to your endpoint
          ↓
5. Your server responds (2xx = success)
          ↓
6. On failure, automatic retry with exponential backoff
```

## Verifying Webhook Signatures

TraceRTM signs all webhook payloads with HMAC-SHA256. Always verify signatures.

### Signature Header

```http
X-TraceRTM-Signature: sha256=abc123def456...
X-TraceRTM-Timestamp: 1702036800
```

### Python Verification

```python
import hmac
import hashlib
import time

def verify_webhook_signature(
    payload: bytes,
    signature: str,
    timestamp: str,
    secret: str,
    tolerance_seconds: int = 300
) -> bool:
    """Verify webhook signature and timestamp."""

    # Check timestamp to prevent replay attacks
    current_time = int(time.time())
    webhook_time = int(timestamp)
    if abs(current_time - webhook_time) > tolerance_seconds:
        return False

    # Create signed payload
    signed_payload = f"{timestamp}.{payload.decode()}"

    # Calculate expected signature
    expected = hmac.new(
        secret.encode(),
        signed_payload.encode(),
        hashlib.sha256
    ).hexdigest()

    # Compare signatures (constant-time)
    return hmac.compare_digest(f"sha256={expected}", signature)


# Flask example
from flask import Flask, request, abort

app = Flask(__name__)
WEBHOOK_SECRET = "whsec_abc123..."

@app.post("/webhook")
def handle_webhook():
    payload = request.get_data()
    signature = request.headers.get("X-TraceRTM-Signature")
    timestamp = request.headers.get("X-TraceRTM-Timestamp")

    if not verify_webhook_signature(payload, signature, timestamp, WEBHOOK_SECRET):
        abort(401, "Invalid signature")

    event = request.get_json()

    # Process event based on type
    if event["event"] == "item.approved":
        handle_item_approved(event["data"])
    elif event["event"] == "test_run.completed":
        handle_test_completed(event["data"])

    return {"status": "received"}, 200
```

### TypeScript Verification

```typescript
import crypto from 'crypto';
import express from 'express';

const WEBHOOK_SECRET = 'whsec_abc123...';

function verifyWebhookSignature(
  payload: string,
  signature: string,
  timestamp: string,
  secret: string,
  toleranceSeconds: number = 300
): boolean {
  // Check timestamp
  const currentTime = Math.floor(Date.now() / 1000);
  const webhookTime = parseInt(timestamp, 10);
  if (Math.abs(currentTime - webhookTime) > toleranceSeconds) {
    return false;
  }

  // Calculate expected signature
  const signedPayload = `${timestamp}.${payload}`;
  const expected = crypto
    .createHmac('sha256', secret)
    .update(signedPayload)
    .digest('hex');

  // Compare signatures (constant-time)
  return crypto.timingSafeEqual(
    Buffer.from(`sha256=${expected}`),
    Buffer.from(signature)
  );
}

const app = express();
app.use(express.raw({ type: 'application/json' }));

app.post('/webhook', (req, res) => {
  const payload = req.body.toString();
  const signature = req.headers['x-tracertm-signature'] as string;
  const timestamp = req.headers['x-tracertm-timestamp'] as string;

  if (!verifyWebhookSignature(payload, signature, timestamp, WEBHOOK_SECRET)) {
    return res.status(401).json({ error: 'Invalid signature' });
  }

  const event = JSON.parse(payload);

  // Process event
  switch (event.event) {
    case 'item.approved':
      handleItemApproved(event.data);
      break;
    case 'test_run.completed':
      handleTestCompleted(event.data);
      break;
  }

  res.status(200).json({ status: 'received' });
});
```

### Go Verification

```go
package main

import (
    "crypto/hmac"
    "crypto/sha256"
    "encoding/hex"
    "fmt"
    "math"
    "strconv"
    "time"
)

func verifyWebhookSignature(payload, signature, timestamp, secret string, toleranceSecs int64) bool {
    // Check timestamp
    webhookTime, _ := strconv.ParseInt(timestamp, 10, 64)
    currentTime := time.Now().Unix()
    if int64(math.Abs(float64(currentTime-webhookTime))) > toleranceSecs {
        return false
    }

    // Calculate expected signature
    signedPayload := fmt.Sprintf("%s.%s", timestamp, payload)
    mac := hmac.New(sha256.New, []byte(secret))
    mac.Write([]byte(signedPayload))
    expected := hex.EncodeToString(mac.Sum(nil))

    // Compare signatures
    return hmac.Equal([]byte("sha256="+expected), []byte(signature))
}
```

## SDK Examples

### Python

```python
from tracertm import TraceRTMClient

client = TraceRTMClient(api_key="rtm_live_sk_...")

# List webhooks
webhooks = client.webhooks.list(status="active")
for wh in webhooks.data:
    print(f"{wh.name}: {wh.statistics.success_rate}% success")

# Create webhook
webhook = client.webhooks.create(
    name="Slack Notifications",
    url="https://hooks.slack.com/services/xxx/yyy/zzz",
    events=["item.approved", "item.rejected"],
    filters={
        "project_ids": ["proj_abc123"],
        "item_types": ["requirement"]
    }
)
print(f"Created: {webhook.id}")
print(f"Secret: {webhook.secret}")  # Store securely!

# Update webhook
client.webhooks.update(
    webhook.id,
    status="paused",
    events=["item.approved", "item.rejected", "item.created"]
)

# Test webhook
result = client.webhooks.test(webhook.id, event="item.approved")
if result.success:
    print(f"Test successful: {result.response_code}")
else:
    print(f"Test failed: {result.error}")

# View delivery history
deliveries = client.webhooks.list_deliveries(
    webhook.id,
    status="failed",
    limit=10
)
for delivery in deliveries.data:
    print(f"{delivery.event}: {delivery.error}")

# Retry failed delivery
client.webhooks.retry_delivery(webhook.id, delivery.id)

# Delete webhook
client.webhooks.delete(webhook.id)
```

## Retry Policy

Failed deliveries are automatically retried with exponential backoff:

| Attempt | Delay |
|---------|-------|
| 1 | Immediate |
| 2 | 1 minute |
| 3 | 5 minutes |
| 4 | 30 minutes |
| 5 | 1 hour |

### Custom Retry Policy

```json
{
  "retry_policy": {
    "max_attempts": 5,
    "initial_delay_seconds": 60,
    "max_delay_seconds": 3600,
    "backoff_multiplier": 2.0
  }
}
```

### Automatic Disabling

Webhooks are automatically disabled after:
- 10 consecutive failures
- 7 days of failures

Re-enable manually after fixing the endpoint.

## Filters

Filter which events trigger the webhook:

```json
{
  "filters": {
    "project_ids": ["proj_abc123", "proj_def456"],
    "item_types": ["requirement", "test_case"],
    "statuses": ["approved", "rejected"],
    "priorities": ["critical", "high"],
    "user_ids": ["user_reviewer123"]
  }
}
```

### Filter Behavior

- Multiple values in a filter: OR (any match)
- Multiple filters: AND (all must match)
- Empty filter: No restriction

## Best Practices

### 1. Always Verify Signatures

```python
# Never skip signature verification
if not verify_signature(payload, signature, secret):
    return 401
```

### 2. Respond Quickly

```python
# Good: Acknowledge quickly, process async
@app.post("/webhook")
def webhook():
    queue.enqueue(process_event, request.json)
    return {"status": "received"}, 200

# Bad: Long processing blocks response
@app.post("/webhook")
def webhook():
    slow_processing(request.json)  # May timeout
    return {"status": "done"}, 200
```

### 3. Handle Duplicates

```python
# Events may be delivered multiple times
processed_events = set()

def handle_event(event):
    if event["id"] in processed_events:
        return  # Already processed
    processed_events.add(event["id"])
    # Process event...
```

### 4. Use Specific Event Filters

```python
# Good: Subscribe to specific events
events = ["item.approved", "item.rejected"]

# Bad: Subscribe to all events
events = ["*"]  # High volume, may miss important events
```
