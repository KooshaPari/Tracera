## Project Setup

### Create a Complete Project

```python
from tracertm import TraceRTMClient

client = TraceRTMClient()

def create_project_with_structure():
    """Create a project with item types and workflows."""

    # Create project
    project = client.projects.create(
        name="Payment System",
        key="PAY",
        description="Payment processing system requirements"
    )
    print(f"Created project: {project.key}")

    # Configure item types
    item_types = [
        {"key": "REQ", "name": "Requirement", "icon": "document"},
        {"key": "TC", "name": "Test Case", "icon": "check"},
        {"key": "BUG", "name": "Bug", "icon": "bug"},
        {"key": "RISK", "name": "Risk", "icon": "warning"}
    ]

    for item_type in item_types:
        client.projects.add_item_type(project.id, **item_type)

    # Configure link types
    link_types = [
        {"key": "verifies", "name": "Verifies", "source": "TC", "target": "REQ"},
        {"key": "mitigates", "name": "Mitigates", "source": "REQ", "target": "RISK"},
        {"key": "blocks", "name": "Blocks", "source": "BUG", "target": "REQ"}
    ]

    for link_type in link_types:
        client.projects.add_link_type(project.id, **link_type)

    return project

project = create_project_with_structure()
```

### Clone Project Structure

```python
def clone_project_structure(source_project_id: str, new_name: str, new_key: str):
    """Clone a project's structure without items."""

    source = client.projects.get(source_project_id)

    # Create new project
    new_project = client.projects.create(
        name=new_name,
        key=new_key,
        description=f"Cloned from {source.name}"
    )

    # Copy workflows
    workflows = client.workflows.list(project_id=source.id)
    for wf in workflows.data:
        client.workflows.copy(
            workflow_id=wf.id,
            target_project_id=new_project.id
        )

    # Copy custom fields
    fields = client.projects.list_custom_fields(source.id)
    for field in fields.data:
        client.projects.add_custom_field(
            project_id=new_project.id,
            name=field.name,
            field_type=field.field_type,
            options=field.options
        )

    return new_project
```

### Import Requirements from CSV

```python
import csv
from pathlib import Path

def import_requirements_from_csv(project_id: str, csv_path: Path):
    """Import requirements from a CSV file."""

    imported = []
    errors = []

    with open(csv_path, newline='', encoding='utf-8') as f:
        reader = csv.DictReader(f)

        for row in reader:
            try:
                item = client.items.create(
                    project_id=project_id,
                    type="requirement",
                    title=row['title'],
                    description=row.get('description', ''),
                    priority=row.get('priority', 'medium'),
                    custom_fields={
                        'external_id': row.get('external_id'),
                        'source': row.get('source', 'CSV Import')
                    }
                )
                imported.append(item)
                print(f"Imported: {item.key}")
            except Exception as e:
                errors.append({'row': row, 'error': str(e)})

    print(f"\nImported {len(imported)} requirements")
    if errors:
        print(f"Errors: {len(errors)}")

    return imported, errors

# Usage
imported, errors = import_requirements_from_csv(
    "proj_abc123",
    Path("./requirements.csv")
)
```

### Export Requirements to JSON

```python
import json
from datetime import datetime

def export_requirements_to_json(project_id: str, output_path: Path):
    """Export all requirements to JSON."""

    requirements = []

    for item in client.items.iterate(project_id=project_id, type="requirement"):
        requirements.append({
            'id': item.id,
            'key': item.key,
            'title': item.title,
            'description': item.description,
            'status': item.status,
            'priority': item.priority,
            'created_at': item.created_at.isoformat(),
            'updated_at': item.updated_at.isoformat(),
            'custom_fields': item.custom_fields
        })

    output = {
        'exported_at': datetime.now().isoformat(),
        'project_id': project_id,
        'count': len(requirements),
        'requirements': requirements
    }

    with open(output_path, 'w') as f:
        json.dump(output, f, indent=2)

    print(f"Exported {len(requirements)} requirements to {output_path}")
    return requirements
```

```python
from datetime import datetime

def record_test_execution(test_case_id: str, result: str, notes: str = None):
    """Record a test execution result."""

    # Create test run if needed
    test_run = client.test_runs.create(
        project_id="proj_abc123",
        name=f"Test Run {datetime.now().strftime('%Y-%m-%d %H:%M')}",

    )

    # Record result
    execution = client.test_executions.create(
        test_run_id=test_run.id,
        test_case_id=test_case_id,
        result=result,  # 'passed', 'failed', 'blocked', 'skipped'

        executed_by="current_user",
        notes=notes,
        duration_seconds=120

    if result == 'failed':
        client.items.transition(
            item_id=test_case_id,
            transition="fail",
            comment=f"Test failed: {notes}"
        )

    return execution
```

### Generate Test Coverage Report

```python
def get_test_coverage(project_id: str):
    """Calculate test coverage for requirements."""

    # Get all requirements
    requirements = list(client.items.iterate(
        project_id=project_id,
        type="requirement"
    ))

    coverage_data = []

    for req in requirements:
        # Get linked test cases
        links = client.links.list(
            source_id=req.id,
            link_type="verified_by"
        )

        test_cases = []
        for link in links.data:
            tc = client.items.get(link.target_id)
            test_cases.append({
                'key': tc.key,
                'title': tc.title,
                'status': tc.status
            })

        coverage_data.append({
            'requirement_key': req.key,
            'requirement_title': req.title,
            'priority': req.priority,
            'test_case_count': len(test_cases),
            'test_cases': test_cases,
            'covered': len(test_cases) > 0
        })

    # Calculate summary
    total = len(coverage_data)
    covered = sum(1 for r in coverage_data if r['covered'])
    coverage_percent = (covered / total * 100) if total > 0 else 0

    return {
        'total_requirements': total,
        'covered_requirements': covered,
        'coverage_percent': round(coverage_percent, 2),
        'details': coverage_data
    }

# Usage
coverage = get_test_coverage("proj_abc123")
print(f"Test Coverage: {coverage['coverage_percent']}%")
```

## Traceability Links

### Build Complete Trace

```python
def build_trace_chain(item_id: str, direction: str = "both"):
    """Build a complete trace chain for an item."""

    item = client.items.get(item_id)

    trace = {
        'item': {
            'id': item.id,
            'key': item.key,
            'title': item.title,
            'type': item.type
        },
        'upstream': [],   # Items this traces from
        'downstream': []  # Items this traces to
    }

    if direction in ("both", "upstream"):
        # Get items this item traces from
        upstream_links = client.links.list(target_id=item.id)
        for link in upstream_links.data:
            source = client.items.get(link.source_id)
            trace['upstream'].append({
                'id': source.id,
                'key': source.key,
                'title': source.title,
                'type': source.type,
                'link_type': link.link_type
            })

    if direction in ("both", "downstream"):
        # Get items this item traces to
        downstream_links = client.links.list(source_id=item.id)
        for link in downstream_links.data:
            target = client.items.get(link.target_id)
            trace['downstream'].append({
                'id': target.id,
                'key': target.key,
                'title': target.title,
                'type': target.type,
                'link_type': link.link_type
            })

    return trace

# Usage
trace = build_trace_chain("item_req_001")
print(f"Item: {trace['item']['key']}")
print(f"Upstream: {len(trace['upstream'])} items")
print(f"Downstream: {len(trace['downstream'])} items")
```

### Find Orphan Items

```python
def find_orphan_requirements(project_id: str):
    """Find requirements with no linked test cases."""

    orphans = []

    for req in client.items.iterate(project_id=project_id, type="requirement"):
        links = client.links.list(
            source_id=req.id,
            link_type="verified_by"
        )

        if len(links.data) == 0:
            orphans.append({
                'key': req.key,
                'title': req.title,
                'priority': req.priority,
                'status': req.status
            })

    # Sort by priority
    priority_order = {'critical': 0, 'high': 1, 'medium': 2, 'low': 3}
    orphans.sort(key=lambda x: priority_order.get(x['priority'], 4))

    return orphans

# Usage
orphans = find_orphan_requirements("proj_abc123")
print(f"Found {len(orphans)} orphan requirements:")
for orphan in orphans:
    print(f"  [{orphan['priority']}] {orphan['key']}: {orphan['title']}")
```

### Impact Analysis

```python
def analyze_impact(item_id: str, max_depth: int = 3):
    """Analyze the impact of changing an item."""

    visited = set()
    impact = []

    def traverse(current_id: str, depth: int, path: list):
        if depth > max_depth or current_id in visited:
            return

        visited.add(current_id)
        item = client.items.get(current_id)

        impact.append({
            'id': item.id,
            'key': item.key,
            'title': item.title,
            'type': item.type,
            'depth': depth,
            'path': ' -> '.join(path + [item.key])
        })

        # Get downstream items
        links = client.links.list(source_id=current_id)
        for link in links.data:
            traverse(link.target_id, depth + 1, path + [item.key])

    traverse(item_id, 0, [])

    return {
        'root_item': item_id,
        'total_impacted': len(impact) - 1,  # Exclude root
        'max_depth_reached': max(i['depth'] for i in impact),
        'impacted_items': impact[1:]  # Exclude root
    }

# Usage
impact = analyze_impact("item_req_001")
print(f"Impact Analysis: {impact['total_impacted']} items affected")
```

### Process Items in Parallel

```python
from concurrent.futures import ThreadPoolExecutor, as_completed

def process_items_parallel(project_id: str, processor_func, max_workers: int = 5):
    """Process items in parallel using thread pool."""

    items = list(client.items.iterate(project_id=project_id))
    results = []
    errors = []

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        future_to_item = {
            executor.submit(processor_func, item): item
            for item in items
        }

        for future in as_completed(future_to_item):
            item = future_to_item[future]
            try:
                result = future.result()
                results.append(result)
            except Exception as e:
                errors.append({'item': item.key, 'error': str(e)})

    return results, errors

# Usage
def enrich_item(item):
    """Add computed fields to item."""
    links = client.links.list(source_id=item.id)
    return {
        'key': item.key,
        'link_count': len(links.data)
    }

results, errors = process_items_parallel("proj_abc123", enrich_item)
```

### Generate Traceability Matrix

```python
def generate_traceability_matrix(project_id: str):
    """Generate a requirements-to-tests traceability matrix."""

    requirements = list(client.items.iterate(
        project_id=project_id,
        type="requirement"
    ))

    test_cases = list(client.items.iterate(
        project_id=project_id,
        type="test_case"
    ))

    # Build matrix
    matrix = []

    for req in requirements:
        row = {
            'requirement_key': req.key,
            'requirement_title': req.title,
            'test_cases': []
        }

        links = client.links.list(source_id=req.id, link_type="verified_by")
        linked_tc_ids = {link.target_id for link in links.data}

        for tc in test_cases:
            row['test_cases'].append({
                'key': tc.key,
                'linked': tc.id in linked_tc_ids
            })

        matrix.append(row)

    return {
        'requirements': [r.key for r in requirements],
        'test_cases': [t.key for t in test_cases],
        'matrix': matrix
    }
```

## Automation

### CI/CD Integration

```python
import sys

def ci_traceability_check(project_id: str):
    """CI pipeline check for traceability compliance."""

    issues = []

    # Check for orphan requirements
    for req in client.items.iterate(project_id=project_id, type="requirement"):
        if req.priority in ('critical', 'high'):
            links = client.links.list(source_id=req.id, link_type="verified_by")
            if len(links.data) == 0:
                issues.append(f"HIGH: {req.key} has no test coverage")

    # Check for failed test cases linked to approved requirements
    for tc in client.items.iterate(project_id=project_id, type="test_case"):
        if tc.status == 'failed':
            links = client.links.list(target_id=tc.id, link_type="verified_by")
            for link in links.data:
                req = client.items.get(link.source_id)
                if req.status == 'approved':
                    issues.append(f"CRITICAL: {tc.key} failed but {req.key} is approved")

    # Report results
    if issues:
        print("Traceability Check FAILED:")
        for issue in issues:
            print(f"  - {issue}")
        sys.exit(1)
    else:
        print("Traceability Check PASSED")
        sys.exit(0)

# Run in CI
if __name__ == "__main__":
    ci_traceability_check("proj_abc123")
```

### Webhook Handler

```python
from flask import Flask, request, jsonify

app = Flask(__name__)
client = TraceRTMClient()

@app.route('/webhook/item-updated', methods=['POST'])
def handle_item_updated():
    """Handle item update webhook."""

    payload = request.json
    item_id = payload['item']['id']
    changes = payload['changes']

    # If requirement status changed to approved
    if payload['item']['type'] == 'requirement':
        if 'status' in changes and changes['status']['new'] == 'approved':
            # Create placeholder test case if none exists
            links = client.links.list(source_id=item_id, link_type="verified_by")
            if len(links.data) == 0:
                item = client.items.get(item_id)
                tc = client.items.create(
                    project_id=item.project_id,
                    type="test_case",
                    title=f"[TODO] Test: {item.title}",
                    description="Auto-generated test case placeholder",
                    priority=item.priority
                )
                client.links.create(
                    source_id=item_id,
                    target_id=tc.id,
                    link_type="verified_by"
                )

    return jsonify({'status': 'processed'})
```
