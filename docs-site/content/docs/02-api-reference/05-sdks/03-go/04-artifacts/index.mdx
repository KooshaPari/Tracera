# Artifacts

The TraceRTM Go SDK provides comprehensive support for managing file artifacts such as documents, images, test results, and other files associated with your traceability items.

## Overview

Artifacts are files attached to items or projects in TraceRTM. Common use cases include:

- Attaching specification documents to requirements
- Uploading test reports to test cases
- Storing design mockups with user stories
- Associating code snippets or logs with bugs

---

## Uploading Artifacts

### Upload from File Path

```go
package main

import (
    "context"
    "fmt"
    "log"

    "github.com/tracertm/tracertm-go"
)

func main() {
    client := tracertm.NewClient(os.Getenv("TRACERTM_API_KEY"))
    ctx := context.Background()

    // Upload a file from disk
    artifact, err := client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
        ItemID:      "item_req_001",
        FilePath:    "./specs/requirements.pdf",
        Description: "Requirements specification document",
    })
    if err != nil {
        log.Fatal(err)
    }

    fmt.Printf("Uploaded: %s\n", artifact.Filename)
    fmt.Printf("Size: %d bytes\n", artifact.Size)
    fmt.Printf("URL: %s\n", artifact.DownloadURL)
}
```

### Upload from Reader

```go
import (
    "bytes"
    "context"
    "io"
)

// Upload from io.Reader (memory efficient for large files)
func uploadFromReader(ctx context.Context, client *tracertm.Client, itemID string, reader io.Reader) (*tracertm.Artifact, error) {
    return client.Artifacts.UploadReader(ctx, &tracertm.UploadReaderInput{
        ItemID:      itemID,
        Reader:      reader,
        Filename:    "report.pdf",
        ContentType: "application/pdf",
        Description: "Generated report",
    })
}

// Upload from bytes
func uploadFromBytes(ctx context.Context, client *tracertm.Client, itemID string, data []byte) (*tracertm.Artifact, error) {
    return client.Artifacts.UploadReader(ctx, &tracertm.UploadReaderInput{
        ItemID:      itemID,
        Reader:      bytes.NewReader(data),
        Filename:    "data.json",
        ContentType: "application/json",
    })
}
```

### Upload from File Handle

```go
import "os"

func uploadFromFile(ctx context.Context, client *tracertm.Client, itemID string, filePath string) (*tracertm.Artifact, error) {
    file, err := os.Open(filePath)
    if err != nil {
        return nil, err
    }
    defer file.Close()

    stat, _ := file.Stat()

    return client.Artifacts.UploadReader(ctx, &tracertm.UploadReaderInput{
        ItemID:      itemID,
        Reader:      file,
        Filename:    stat.Name(),
        Size:        stat.Size(),
        ContentType: detectContentType(filePath),
    })
}
```

### Stream Large Files

```go
import "io"

// Stream large files to avoid memory issues
func streamToFile(ctx context.Context, client *tracertm.Client, artifactID, destPath string) error {
    reader, err := client.Artifacts.DownloadStream(ctx, artifactID)
    if err != nil {
        return err
    }
    defer reader.Close()

    file, err := os.Create(destPath)
    if err != nil {
        return err
    }
    defer file.Close()

    _, err = io.Copy(file, reader)
    return err
}
```

### Get Download URL

```go
    ExpiresIn: 3600, // URL valid for 1 hour

    log.Fatal(err)
}
```

## Listing Artifacts

### List Item Artifacts

```go
// List all artifacts for an item
artifacts, err := client.Artifacts.List(ctx, &tracertm.ListArtifactsInput{
    ItemID: "item_req_001",
})
if err != nil {
    log.Fatal(err)
}

for _, artifact := range artifacts.Data {
    fmt.Printf("%s (%d bytes)\n", artifact.Filename, artifact.Size)
    fmt.Printf("  Uploaded: %s\n", artifact.CreatedAt)
    fmt.Printf("  Type: %s\n", artifact.ContentType)
}
```

### List Project Artifacts

```go
// List all artifacts in a project
artifacts, err := client.Artifacts.List(ctx, &tracertm.ListArtifactsInput{
    ProjectID: "proj_abc123",
    Limit:     50,
    Offset:    0,
})
```

### Filter Artifacts

```go
// Filter by content type
artifacts, err := client.Artifacts.List(ctx, &tracertm.ListArtifactsInput{
    ProjectID:   "proj_abc123",
    ContentType: "application/pdf",
})

// Filter by filename pattern
artifacts, err := client.Artifacts.List(ctx, &tracertm.ListArtifactsInput{
    ProjectID:        "proj_abc123",
    FilenameContains: "report",
})

// Filter by date range
artifacts, err := client.Artifacts.List(ctx, &tracertm.ListArtifactsInput{
    ProjectID:    "proj_abc123",
    CreatedAfter: time.Now().Add(-7 * 24 * time.Hour), // Last 7 days
})
```

### Iterate All Artifacts

```go
// Iterate through all artifacts (handles pagination)
iter := client.Artifacts.Iterate(ctx, &tracertm.ListArtifactsInput{
    ProjectID: "proj_abc123",
})

for iter.Next() {
    artifact := iter.Artifact()
    fmt.Printf("%s: %d bytes\n", artifact.Filename, artifact.Size)
}

if err := iter.Err(); err != nil {
    log.Fatal(err)
}
```

## Bulk Operations

### Bulk Upload

```go
import "path/filepath"

// Upload multiple files at once
func bulkUpload(ctx context.Context, client *tracertm.Client, itemID string, dir string) ([]*tracertm.Artifact, error) {
    files, err := filepath.Glob(filepath.Join(dir, "*.pdf"))
    if err != nil {
        return nil, err
    }

    var artifacts []*tracertm.Artifact
    for _, file := range files {
        artifact, err := client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
            ItemID:      itemID,
            FilePath:    file,
            Description: "Specification document",
        })
        if err != nil {
            return artifacts, err
        }
        artifacts = append(artifacts, artifact)
        fmt.Printf("Uploaded: %s\n", artifact.Filename)
    }

    return artifacts, nil
}
```

### Bulk Download

```go
// Download all artifacts for an item
func bulkDownload(ctx context.Context, client *tracertm.Client, itemID, destDir string) error {
    iter := client.Artifacts.Iterate(ctx, &tracertm.ListArtifactsInput{
        ItemID: itemID,
    })

    for iter.Next() {
        artifact := iter.Artifact()
        destPath := filepath.Join(destDir, artifact.Filename)

        if err := client.Artifacts.DownloadToFile(ctx, artifact.ID, destPath); err != nil {
            return err
        }
        fmt.Printf("Downloaded: %s\n", artifact.Filename)
    }

    return iter.Err()
}
```

## Concurrent Operations

### Parallel Uploads

```go
import "sync"

func parallelUpload(ctx context.Context, client *tracertm.Client, itemID string, files []string, workers int) ([]*tracertm.Artifact, []error) {
    var (
        wg        sync.WaitGroup
        mu        sync.Mutex
        artifacts []*tracertm.Artifact
        errors    []error
    )

    sem := make(chan struct{}, workers)

    for _, file := range files {
        wg.Add(1)
        go func(filePath string) {
            defer wg.Done()
            sem <- struct{}{}
            defer func() { <-sem }()

            artifact, err := client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
                ItemID:   itemID,
                FilePath: filePath,
            })

            mu.Lock()
            if err != nil {
                errors = append(errors, err)
            } else {
                artifacts = append(artifacts, artifact)
            }
            mu.Unlock()
        }(file)
    }

    wg.Wait()
    return artifacts, errors
}
```

### Context Cancellation

```go
func uploadWithTimeout(client *tracertm.Client, itemID, filePath string) (*tracertm.Artifact, error) {
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Minute)
    defer cancel()

    return client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
        ItemID:   itemID,
        FilePath: filePath,
    })
}
```

## Error Handling

```go
import "errors"

artifact, err := client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
    ItemID:   "item_req_001",
    FilePath: "./large-file.zip",
})

if err != nil {
    var fileTooLargeErr *tracertm.FileTooLargeError
    var unsupportedTypeErr *tracertm.UnsupportedContentTypeError
    var quotaExceededErr *tracertm.QuotaExceededError
    var notFoundErr *tracertm.NotFoundError
    var validationErr *tracertm.ValidationError

    switch {
    case errors.As(err, &fileTooLargeErr):
        fmt.Printf("File too large: %d bytes (max: %d)\n", fileTooLargeErr.Size, fileTooLargeErr.MaxSize)

    case errors.As(err, &unsupportedTypeErr):
        fmt.Printf("Unsupported type: %s\n", unsupportedTypeErr.ContentType)

    case errors.As(err, &quotaExceededErr):
        fmt.Printf("Storage quota exceeded: %d/%d bytes\n", quotaExceededErr.Used, quotaExceededErr.Limit)

    case errors.As(err, &notFoundErr):
        fmt.Println("Item not found")

    case errors.As(err, &validationErr):
        for _, e := range validationErr.Errors {
            fmt.Printf("  %s: %s\n", e.Field, e.Message)
        }

    default:
        fmt.Printf("Error: %v\n", err)
    }
}
```

## Best Practices

### 1. Use Streaming for Large Files

```go
// Bad - loads entire file into memory
content, _ := os.ReadFile("./large-file.zip")
client.Artifacts.UploadReader(ctx, &tracertm.UploadReaderInput{
    ItemID: itemID,
    Reader: bytes.NewReader(content),
    ...
})

// Good - streams the file
client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
    ItemID:   itemID,
    FilePath: "./large-file.zip",
})
```

### 2. Validate Before Upload

```go
func safeUpload(ctx context.Context, client *tracertm.Client, itemID, filePath string) (*tracertm.Artifact, error) {
    stat, err := os.Stat(filePath)
    if err != nil {
        return nil, fmt.Errorf("file not found: %s", filePath)
    }

    maxSize := int64(100 * 1024 * 1024) // 100 MB
    if stat.Size() > maxSize {
        return nil, fmt.Errorf("file too large: %s", filePath)
    }

    return client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
        ItemID:   itemID,
        FilePath: filePath,
    })
}
```

### 3. Use Meaningful Descriptions

```go
// Good - descriptive
artifact, err := client.Artifacts.Upload(ctx, &tracertm.UploadArtifactInput{
    ItemID:      "item_req_001",
    FilePath:    "./specs/auth-requirements-v2.pdf",
    Description: "Authentication requirements specification v2.0 - includes OAuth 2.0 and SAML",
})
```
